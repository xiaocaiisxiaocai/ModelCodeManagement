## 🔧 API接口设计（已修复）

### 编码预分配API
```csharp
[ApiController]
[Route("api/v1/[controller]")]  // ✅ 修复：统一使用 /api/v1/ 前缀
[Authorize(Roles = "Admin,SuperAdmin")]
public class CodePreAllocationController : ControllerBase
{
    [HttpPost("allocate-codes")]
    public async Task<ApiResult> PreAllocateCodes(PreAllocateCodesDto dto)
    {
        await _preAllocationService.PreAllocateCodesAsync(
            dto.ModelType, dto.CodeClassification, dto.ClassificationId);
        return ApiResult.Success();
    }
    
    [HttpPost("create-manual-code")]
    public async Task<ApiResult> CreateManualCode(CreateManualCodeDto dto)
    {
        var success = await _preAllocationService.CreateManualCodeAsync(
            dto.ModelType, dto.NumberPart, dto.ExtensionPart, dto);
        if (success)
            return ApiResult.Success();
        else
            return ApiResult.Error("编码格式错误或已存在");
    }
    
    [HttpPost("validate-manual-code")]
    public async Task<ApiResult<bool>> ValidateManualCode(ValidateManualCodeDto dto)
    {
        var isValid = await _preAllocationService.ValidateManualCodeFormatAsync(
            dto.ModelType, dto.NumberPart, dto.ExtensionPart);
        return ApiResult.Success(isValid);
    }
    
    [HttpPost("update-occupancy-type")]
    public async Task<ApiResult> UpdateOccupancyType(UpdateOccupancyTypeDto dto)
    {
        var success = await _preAllocationService.UpdateOccupancyTypeAsync(
            dto.CodeUsageId, dto.OccupancyType);
        
        // 发布占用类型变更事件，状态转换监控逻辑由监控服务统一处理
        if (success)
        {
            await _eventBus.PublishAsync(new OccupancyTypeChangedEvent
            {
                CodeUsageId = dto.CodeUsageId,
                OldOccupancyType = oldOccupancyType,
                NewOccupancyType = dto.OccupancyType,
                Timestamp = DateTime.Now
            });
        }
        
        if (success)
            return ApiResult.Success();
        else
            return ApiResult.Error("更新占用类型失败");
    }
    
    [HttpGet("available-codes/{modelType}/{classificationId}")]
    public async Task<ApiResult<List<AvailableCodeDto>>> GetAvailableCodes(
        string modelType, int classificationId)
    {
        var availableCodes = await _db.Queryable<CodeUsageEntry>()
            .Where(c => c.CodeClassificationId == classificationId 
                       && c.IsAllocated == false 
                       && c.IsDeleted == false)
            .OrderBy(c => c.CodeNumber)
            .Select(c => new AvailableCodeDto
            {
                Id = c.Id,
                Model = c.Model,           // SLU-100
                CodeNumber = c.CodeNumber, // 100
                IsAvailable = true,
                NumberDigits = c.NumberDigits
            })
            .ToListAsync();
            
        return ApiResult.Success(availableCodes);
    }
    
    [HttpPut("use-code/{codeId}")]
    public async Task<ApiResult> UseCode(int codeId, UseCodeDto dto)
    {
        var success = await _preAllocationService.AllocateCodeAsync(codeId, dto);
        if (success)
            return ApiResult.Success();
        else
            return ApiResult.Error("编码不可用或已被使用");
    }
}
```

### 位数升级API
```csharp
[ApiController]
[Route("api/v1/[controller]")]  // ✅ 修复：统一使用 /api/v1/ 前缀
[Authorize(Roles = "SuperAdmin")]
public class SystemConfigController : ControllerBase
{
    [HttpPost("upgrade-number-digits")]
    public async Task<ApiResult> UpgradeNumberDigits(UpgradeDigitsDto dto)
    {
        var currentDigits = await GetCurrentNumberDigitsAsync();
        
        if (dto.NewDigits <= currentDigits)
        {
            return ApiResult.Error("新位数必须大于当前位数");
        }
        
        using var transaction = await _db.Ado.BeginTranAsync();
        try
        {
            // 1. 将旧位数的未使用预分配编码标记为失效
            await _db.Updateable<CodeUsageEntry>()
                .SetColumns(c => new CodeUsageEntry 
                { 
                    IsDeleted = true,
                    UpdatedAt = DateTime.Now,
                    DeletedReason = $"位数升级从{currentDigits}位到{dto.NewDigits}位时自动失效"
                })
                .Where(c => c.NumberDigits == currentDigits 
                           && c.IsAllocated == false 
                           && c.IsDeleted == false)
                .ExecuteCommandAsync();
            
            // 2. 更新系统配置
            await _db.Updateable<SystemConfig>()
                .SetColumns(c => new SystemConfig 
                { 
                    ConfigValue = dto.NewDigits.ToString(),
                    UpdatedAt = DateTime.Now 
                })
                .Where(c => c.ConfigKey == "NumberDigits")
                .ExecuteCommandAsync();
            
            // 3. 为现有的代码分类重新预分配编码（使用新位数）
            await ReAllocateExistingClassifications(dto.NewDigits);
            
            await transaction.CommitAsync();
            return ApiResult.Success($"编号位数已从{currentDigits}位升级到{dto.NewDigits}位，预分配编码已重新生成");
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }
}
```

## 📝 修复说明

### 已修复的问题：
1. **API路径规范统一**：所有控制器现在统一使用 `/api/v1/[controller]` 路径前缀，与项目文档和开发组件图的要求保持一致。

### API路径示例：
- 预分配编码：`POST /api/v1/codepreallocation/allocate-codes`
- 创建手动编码：`POST /api/v1/codepreallocation/create-manual-code`
- 验证编码：`POST /api/v1/codepreallocation/validate-manual-code`
- 获取可用编码：`GET /api/v1/codepreallocation/available-codes/{modelType}/{classificationId}`
- 使用编码：`PUT /api/v1/codepreallocation/use-code/{codeId}`
- 升级位数：`POST /api/v1/systemconfig/upgrade-number-digits`

这样确保了与其他文档中定义的API设计规范完全一致。
